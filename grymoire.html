Then add the lsos() function 

.ls.objects <- function (pos = 1, pattern, order.by,
                        decreasing=FALSE, head=FALSE, n=5) {
    # Function for listing ALL objects and size
    # improved list of objects
    napply <- function(names, fn) sapply(names, function(x)
                                         fn(get(x, pos = pos)))
    names <- ls(pos = pos, pattern = pattern)
    obj.class <- napply(names, function(x) as.character(class(x))[1])
    obj.mode <- napply(names, mode)
    obj.type <- ifelse(is.na(obj.class), obj.mode, obj.class)
    obj.prettysize <- napply(names, function(x) {
                           capture.output(print(object.size(x), units = "auto")) })
    obj.size <- napply(names, object.size)
    obj.dim <- t(napply(names, function(x)
                        as.numeric(dim(x))[1:2]))
    vec <- is.na(obj.dim)[, 1] & (obj.type != "function")
    obj.dim[vec, 1] <- napply(names, length)[vec]
    out <- data.frame(obj.type, obj.size, obj.prettysize, obj.dim)
    names(out) <- c("Type", "Size", "PrettySize", "Rows", "Columns")
    if (!missing(order.by))
        out <- out[order(out[[order.by]], decreasing=decreasing), ]
    if (head)
        out <- head(out, n)
    out
}

lsos <- function(..., n=100) {
    # shorthand
    .ls.objects(..., decreasing=TRUE, head=TRUE, n=n)   # order.by="Size"
}

PREINSTALLATION AND CONFIGURATION OF ARCHLINUX (EASY INSTALL)

Keyboard layout. List layouts and choose French layout. 

less /usr/share/kbd/keymaps/
localectl list-keymaps
loadkeys fr

GPT partitions 

gdisk /dev/sda

create partitions using p:print, d:delete, w:write

Description of partitions Partitions
BIOS boot partition +84K (code ef02) sda1
Boot partition +350M sda2
Swap +2-4G (code 8200) sda3
/var +8G sda4
/home +40G sda5
/root +60G sda6

Formatting Boot. But should leave BIOS untouched. ext2 dont have journalling which is why ext4 is better for root. 

mkfs.ext2 /dev/sda2
mkfs.ext4 /dev/sda6
mkfs.xfs /dev/sda5
mkswap /dev/sda3
swapon /dev/sda3
mkreiserfs /dev/sda4

Mounting, mount root. mount root before mkdir 

mount /dev/sda6 /mnt
cd /mnt && mkdir home boot var
mount /dev/sda2 /mnt/boot
mount /dev/sda4 /mnt/var
mount /dev/sda5 /mnt/home

Tree visualization of sdas 

lsblk -f

pacstrap -i /mnt base base-devel

generate fstab file 

genfstab -L -p /mnt >> /mnt/etc/fstab
nano /mnt/etc/fstab

change root

setup and test the internet after drivers configuration. Generate file for customizations. 

pacman -S net-tools iw wireless_tools
pacman -S dialog wpa_actiond ifplugd wpa_suppicant sudo zsh

install emacs 

pacman -S emacs

Language. uncomment enUS and generate file for configuations. 

emacs /etc/locale.gen
locale-gen
echo LANG=en_US.UTF-8 > /etc/locale.conf
export LANG=en_US.UTF-8

Create user 

useradd -m -g users -G wheel,audio,video,optical,storage,power,lp,locate -s /usr/bin/zsh neo

uncomment "# %wheel ALL=(ALL) ALL" from sudo, and make zsh the default shell for root too 

emacs sudo
chsh -s /usr/bin/zsh

TimeZone. Set to US vermont and the configure the hardware clock. 

ls /usr/share/zoneinfo
ln -sf /usr/share/zoneinfo/US/eastern /etc/localtime
hwclock --systohc --utc

Hostname, Replace hostname-domain with arch after the nano command 

echo arch > /etc/hostname
PC ID:arch1
nano /etc/hosts

Fix configs and set user password

passwd neo

GRUB, Download / install GRUB. Customize grub: Replace "quiet" with vga=791 quiet then uncomment GRUBCOLORNORMAL, GRUBCOLORHIGHLIGHT, GRUBINITTUNE. 

pacman -S grub
grub-install --target=i386-pc --recheck /dev/sda
emacs /etc/default/grub
grub-mkconfig /dev/sda -o /boot/grub/grub.cfg

Finish 

exit
reboot
